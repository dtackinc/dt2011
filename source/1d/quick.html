Quick Sort

/* quick sort */

void sort (int *box, int length)
{
	quick (box, 0, length-1);
}

void quick (int *rox, int left, int right)
{
	int centre, i, j;
	extern int comparecount;

	if (left < right) {
		centre = rox[(left + right) / 2];
		i = left - 1;
		j = right + 1;
		for (;;) {
			while (rox[++i] < centre)
				comparecount++;
			while (rox[--j] > centre)
				comparecount++;
			if (i >= j)
				break;
			if (rox[i] != rox[j])
				swap (&rox[i], &rox[j]);
		}
		quick (rox, left, i - 1);
		quick (rox, j + 1, right);
	}
}
