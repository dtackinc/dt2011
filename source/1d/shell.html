/* shell sort */

void sort (int *box, int length)
{
	int gap, a, b;
	extern int comparecount;
	for (gap = 1; gap < length / 3; gap = 3 * gap + 1);
	while (gap > 0) {
		for (a = gap; a < length; a++)
			for (b = a - gap; b >= 0; b -= gap) {
				comparecount++;
				if (box[b] > box[b+gap])
					swap (&box[b], &box[b+gap]);
			}
		gap /= 3;
	}
}
