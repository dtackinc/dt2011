/* shaker sort */

void sort (int *rox, int length)
{
	int n, i, j;
	extern int comparecount;
	for (n = 1; n < length/2; n++) {
		for (i = n; i < length - n; i++) {
			comparecount++;
			if (rox[i] > rox[i+1])
				swap (&rox[i], &rox[i+1]);
		}
		for (j = length - n; j >= n; j--) {
			comparecount++;
			if (rox[j] < rox[j-1])
				swap (&rox[j], &rox[j-1]);
		}
	}
}
