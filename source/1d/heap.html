/* heap sort */

void shift (int *rox, int parent, int num)
{
	int son;
	extern comparecount;
	son = 2 * parent;
	while (son <= num) {
		comparecount++;
		if (son < num && rox[son-1] < rox[son])
			son++;
		comparecount++;
		if (rox[parent-1] >= rox[son-1])
			break;
		swap (&rox[parent-1], &rox[son-1]);
		parent = son;
		son = 2 * parent;
	}
}

void sort (int *box, int length)
{
	int l, j;
	l = length;
	for (j = (l/2); j >= 1 ; j--)
		shift (box, j, l);
	while (l > 1) {
		swap (&box[0], &box[l-1]);
		shift (box, 1, --l);
	}
}
