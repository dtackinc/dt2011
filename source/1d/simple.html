/* simple sort */

void sort (int *box, int length)
{
	int n, i, num;
	extern int comparecount;

	for (n = 0; n < length ; n++) {
		num = n;
		for (i = n; i < length; i++) {
			comparecount++;
			if (box[num] > box[i])
				num = i;
		}		
		if (num != n)
		swap (&box[num], &box[n]);
	}
}
